class Funktor
{
  	public:
		virtual void execute()=0;
}

template<class T>
class Command: public Funktor
{
	protected:
		Promise<T> prom;
		b::function<T> command;
		
	public:
		void execute()
		{
			prom.setValue(command());
		}
		void setProgress(double p)
		{
			prom.setProgress(p);
		}
}
